Existing methods for insider trading detection typically rely on statistical analysis of trading data, surveillance systems that monitor trading activity, and investigations triggered by regulatory alerts or tips. Some common approaches include:

1.  **Volume and Price Surveillance:** These methods look for unusual changes in trading volume or price movements that precede significant corporate announcements. Statistical techniques, such as event study analysis, are used to determine whether the observed changes are statistically significant and potentially indicative of insider trading.

    *   **Limitations:** These methods can be easily circumvented by insiders who trade in smaller volumes or use more sophisticated trading strategies to avoid detection. They may also generate many false positives due to normal market fluctuations or other factors unrelated to insider trading.

2.  **Insider Reporting Analysis:** Regulatory bodies require insiders to report their trades in the company's securities within a specific timeframe. These reports are analyzed to identify patterns or anomalies that may suggest illegal activity.

    *   **Limitations:** Insiders may fail to report their trades accurately or on time, or they may use nominee accounts to conceal their trading activity. Furthermore, analyzing individual insider reports may not reveal coordinated trading schemes involving multiple individuals.

3.  **Link Analysis:** This technique involves mapping the relationships between individuals and entities involved in trading activity to identify potential connections and collusive behavior. Social network analysis and other graph-based methods can be used to uncover hidden relationships that may not be apparent from analyzing individual trades.

    *   **Limitations:** Link analysis can be computationally intensive and may require access to vast amounts of data. It can also be difficult to distinguish between legitimate relationships and those indicative of illegal activity.

4.  **Machine Learning Techniques:** Various machine learning algorithms, such as classification models and anomaly detection techniques, have been applied to insider trading detection. These models are trained on historical trading data and other relevant information to identify patterns that are indicative of insider trading.

    *   **Limitations:** Machine learning models can be data-hungry and may require large amounts of labeled data to achieve high accuracy. They can also be susceptible to overfitting and may not generalize well to new or unseen data. Furthermore, insiders may adapt their trading strategies to evade detection by these models.

5.  **Clustering Analysis:** This involves grouping insider trades based on various characteristics, such as the timing of the trades, the identity of the insiders, and the size of the trades. The goal is to identify clusters of trades that are statistically significant and potentially indicative of coordinated insider trading activity.

    *   **Limitations:** Traditional clustering methods may not be effective in detecting complex or subtle insider trading schemes. They may also be sensitive to the choice of clustering parameters and may require significant human intervention to interpret the results. Furthermore, they often fail to incorporate contextual information that may be relevant to assessing the likelihood of insider trading.

#!/usr/bin/env python
import os
import subprocess
import pandas as pd
import numpy as np
import google.generativeai as genai
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
from configparser import ConfigParser
from qdrant_client import QdrantClient, models
from qdrant_client.models import VectorParams, Distance, PointStruct

# Configuration
GEMINI_API_KEY = os.environ.get("GEMINI_API_KEY")
QDRANT_HOST = os.environ.get("QDRANT_HOST", "localhost")
QDRANT_PORT = int(os.environ.get("QDRANT_PORT", 6333))
QDRANT_API_KEY = os.environ.get("QDRANT_API_KEY")
COLLECTION_NAME = "insider_trades"

# Gemini Setup
genai.configure(api_key=GEMINI_API_KEY)
model = genai.GenerativeModel('gemini-pro')

# Qdrant Setup
client = QdrantClient(host=QDRANT_HOST, port=QDRANT_PORT, api_key=QDRANT_API_KEY)

def analyze_sentiment(text):
    """Analyzes the sentiment of a given text using Gemini."""
    try:
        response = model.generate_content(f"Analyze the sentiment of the following text: '{text}'. Return 'positive', 'negative', or 'neutral'.")
        sentiment = response.text.strip().lower()
        if sentiment not in ['positive', 'negative', 'neutral']:
            return 'neutral'  # Handle cases where Gemini's response is unexpected
        return sentiment
    except Exception as e:
        print(f"Error analyzing sentiment: {e}")
        return 'neutral'

def extract_keywords(text):
    """Extracts keywords from a given text using Gemini."""
    try:
        response = model.generate_content(f"Extract 5 keywords from the following text: '{text}'. Return them as a comma-separated list.")
        keywords = [k.strip() for k in response.text.split(',')]
        return keywords
    except Exception as e:
        print(f"Error extracting keywords: {e}")
        return []

def load_insider_trades(file_path="data/insider_trades.csv"):
    """Loads insider trades data from a CSV file."""
    try:
        df = pd.read_csv(file_path)
        # Basic data cleaning/validation
        df = df.dropna(subset=['insider_name', 'trade_date', 'trade_volume', 'company_description', 'communication_text'])
        return df
    except FileNotFoundError:
        print(f"Error: File not found at {file_path}")
        return None
    except Exception as e:
        print(f"Error loading insider trades data: {e}")
        return None

def load_news_articles(file_path="data/news_articles.csv"):
    """Loads news articles from a CSV file."""
    try:
        df = pd.read_csv(file_path)
        df = df.dropna(subset=['text'])  # Ensure 'text' column exists
        return df
    except FileNotFoundError:
        print(f"Error: File not found at {file_path}")
        return None
    except Exception as e:
        print(f"Error loading news articles: {e}")
        return None

def create_features(insider_trades, news_articles):
    """Creates features for insider trading detection."""

    # Sentiment analysis of insider communications
    insider_trades['sentiment'] = insider_trades['communication_text'].apply(analyze_sentiment)
    insider_trades['sentiment_score'] = insider_trades['sentiment'].map({'positive': 1, 'negative': -1, 'neutral': 0}).fillna(0)

    # News relevance score (example: count of relevant keywords in news articles)
    def calculate_news_relevance(trade):
        relevant_keywords = extract_keywords(trade['company_description'])
        count = 0
        for _, row in news_articles.iterrows():
            if any(keyword in row['text'].lower() for keyword in relevant_keywords):
                count += 1
        return count

    insider_trades['news_relevance'] = insider_trades.apply(calculate_news_relevance, axis=1).fillna(0)

    # Combine features
    features = insider_trades[['trade_volume', 'sentiment_score', 'news_relevance']].fillna(0)  # Handle potential NaN values
    return features

def cluster_insider_trades(features, n_clusters=5):
    """Clusters insider trades based on their features."""
    scaler = StandardScaler()
    scaled_features = scaler.fit_transform(features)

    kmeans = KMeans(n_clusters=n_clusters, random_state=0, n_init = 'auto')  # Explicitly set n_init
    clusters = kmeans.fit_predict(scaled_features)
    return clusters, scaler

def upload_to_qdrant(insider_trades, features, clusters):
    """Uploads insider trade data and cluster assignments to Qdrant."""
    try:
        client.recreate_collection(
            collection_name=COLLECTION_NAME,
            vectors_config=VectorParams(size=features.shape[1], distance=Distance.COSINE),
        )

        points = []
        for i, row in insider_trades.iterrows():
            vector = features.iloc[i].values.tolist()
            payload = row.to_dict()
            payload['cluster'] = int(clusters[i])  # Ensure cluster is an integer
            points.append(PointStruct(id=i, vector=vector, payload=payload))

        client.upsert(collection_name=COLLECTION_NAME, points=points, wait=True)
        print("Data uploaded to Qdrant successfully.")

    except Exception as e:
        print(f"Error uploading to Qdrant: {e}")

def main():
    """Main function to detect insider trading."""

    # 1. Data Loading
    insider_trades = load_insider_trades()
    if insider_trades is None:
        print("Failed to load insider trades data. Exiting.")
        return

    news_articles = load_news_articles()
    if news_articles is None:
        print("Failed to load news articles. Exiting.")
        return

    # 2. Feature Engineering
    features = create_features(insider_trades, news_articles)

    # 3. Clustering
    clusters, scaler = cluster_insider_trades(features)

    # Add cluster assignments to the insider_trades DataFrame
    insider_trades['cluster'] = clusters

    # 4. Upload to Qdrant
    upload_to_qdrant(insider_trades, features, clusters)

    print("Insider trading detection process completed.")

if __name__ == "__main__":
    # Check if Gemini API key is set
    if not GEMINI_API_KEY:

   # Check if Qdrant API key is set
    if not QDRANT_API_KEY:
        print("Error: QDRANT_API_KEY environment variable not set.")
        print("Please set the QDRANT_API_KEY environment variable and try again.")
        exit(1)

    main()
